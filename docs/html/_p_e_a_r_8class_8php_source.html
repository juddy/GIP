633 function _PEAR_call_destructors()
634 {
635  global $_PEAR_destructor_object_list;
636  if (is_array($_PEAR_destructor_object_list) &&
637  sizeof($_PEAR_destructor_object_list))
638  {
639  reset($_PEAR_destructor_object_list);
640  while (list($k, $objref) = each($_PEAR_destructor_object_list)) {
641  $classname = get_class($objref);
642  while ($classname) {
643  $destructor = "_$classname";
644  if (method_exists($objref, $destructor)) {
645  $objref->$destructor();
646  break;
647  } else {
648  $classname = get_parent_class($classname);
649  }
650  }
651  }
652  // Empty the object list to ensure that destructors are
653  // not called more than once.
654  $_PEAR_destructor_object_list = array();
655  }
656 
657  // Now call the shutdown functions
658  if (is_array($GLOBALS['_PEAR_shutdown_funcs']) AND !empty($GLOBALS['_PEAR_shutdown_funcs'])) {
659  foreach ($GLOBALS['_PEAR_shutdown_funcs'] as $value) {
660  call_user_func_array($value[0], $value[1]);

702  function PEAR_Error($message = 'unknown error', $code = null,
703  $mode = null, $options = null, $userinfo = null)
704  {
705  if ($mode === null) {
706  $mode = PEAR_ERROR_RETURN;
707  }
708  $this->message = $message;
709  $this->code = $code;
710  $this->mode = $mode;
711  $this->userinfo = $userinfo;
712  if (function_exists("debug_backtrace")) {
713  $this->backtrace = debug_backtrace();
714  }
715  if ($mode & PEAR_ERROR_CALLBACK) {
716  $this->level = E_USER_NOTICE;
717  $this->callback = $options;
718  } else {
719  if ($options === null) {
720  $options = E_USER_NOTICE;
721  }
722  $this->level = $options;
723  $this->callback = null;
724  }
725  if ($this->mode & PEAR_ERROR_PRINT) {
726  if (is_null($options) || is_int($options)) {
727  $format = "%s";
728  } else {
729  $format = $options;
730  }
731  printf($format, $this->getMessage());
732  }
733  if ($this->mode & PEAR_ERROR_TRIGGER) {
734  trigger_error($this->getMessage(), $this->level);
735  }
736  if ($this->mode & PEAR_ERROR_DIE) {
737  $msg = $this->getMessage();
738  if (is_null($options) || is_int($options)) {
739  $format = "%s";
740  if (substr($msg, -1) != "\n") {
741  $msg .= "\n";
742  }
743  } else {
744  $format = $options;
745  }
746  die(sprintf($format, $msg));
747  }
748  if ($this->mode & PEAR_ERROR_CALLBACK) {
749  if (is_string($this->callback) && strlen($this->callback)) {
750  call_user_func($this->callback, $this);
751  } elseif (is_array($this->callback) &&
752  sizeof($this->callback) == 2 &&
753  is_object($this->callback[0]) &&
754  is_string($this->callback[1]) &&
755  strlen($this->callback[1])) {
756  @call_user_func($this->callback, $this);
757  }
758  }
759  }
