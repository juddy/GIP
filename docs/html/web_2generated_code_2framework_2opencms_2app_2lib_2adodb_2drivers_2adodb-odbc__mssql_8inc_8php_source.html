182  function SQLDate($fmt, $col=false)
183  {
184  if (!$col) $col = $this->sysTimeStamp;
185  $s = '';
186 
187  $len = strlen($fmt);

191  switch($ch) {
192  case 'Y':
193  case 'y':
194  $s .= "datename(yyyy,$col)";
195  break;
196  case 'M':
197  $s .= "convert(char(3),$col,0)";
198  break;
199  case 'm':
200  $s .= "replace(str(month($col),2),' ','0')";
201  break;
202  case 'Q':
203  case 'q':
204  $s .= "datename(quarter,$col)";
205  break;
206  case 'D':
207  case 'd':
208  $s .= "replace(str(day($col),2),' ','0')";
209  break;
210  case 'h':
211  $s .= "substring(convert(char(14),$col,0),13,2)";
212  break;
213 
214  case 'H':
215  $s .= "replace(str(datepart(hh,$col),2),' ','0')";
216  break;
217 
218  case 'i':
219  $s .= "replace(str(datepart(mi,$col),2),' ','0')";
220  break;
221  case 's':
222  $s .= "replace(str(datepart(ss,$col),2),' ','0')";
223  break;
224  case 'a':
225  case 'A':
226  $s .= "substring(convert(char(19),$col,0),18,2)";
227  break;
228 
229  default:
